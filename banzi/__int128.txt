void print(__int128 x){
    if(x<0){
        putchar('-');
        x=-x;
    }
    if(x>9)print(x/10);
    putchar(x%10+'0');
}
__int128 read(){
    __int128 f=1,w=0;
    char ch=getchar();
    while(ch<'0'||ch>'9'){
        if(ch=='-')
        f=-1;
        ch=getchar();
    }
    while(ch<='9'&&ch>='0'){
        w=w*10+ch-'0';
        ch=getchar();
    }
    return f*w;
}
